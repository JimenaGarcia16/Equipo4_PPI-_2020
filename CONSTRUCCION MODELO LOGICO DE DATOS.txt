/*DIRECTIVOS*/
CREATE TABLE DIRECTIVOS(
    ID_DIRECTIVOS INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    NOMBRE VARCHAR(30) NOT NULL,
    APELLIDO VARCHAR(50),
    CARGO TEXT,
    COD_DIRECTIVOSPK INT DEFAULT 12);

/*ESTUDIANTE/
CREATE TABLE ESTUDIANTE (
    ID_ESTUDIANTESPK INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    NOMBRE_COMPLETO VARCHAR (60) NOT NULL,
    N_ANOTACIONES INT DEFAULT 2,
    CORREO_ELECTRONICO VARCHAR (60) NOT NULL,
